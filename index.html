 * rationnumber.cpp
 * calculates proper fractions without reducing to decimal numbers
 */
class rationalnumber {
private:
  int numerator;
  int denominator;
public:
  void normalize() {
      if (numerator==0) {
          denominator = 1;
      }
      int n = numerator 1 ; i--) {
          if (numerator%i==0&&denominator%i==0) {
              numerator /= i;
              denominator /= i;
              normalize();
          }
      }
      if (denominator<0) {
          numerator = -numerator;
          denominator = -denominator;
      }
  }

  rationalnumber() {
      numerator = 0;
      denominator = 1;
  }

  rationalnumber(int n, int d) {
      numerator = n;
      denominator = d;
  }

  rationalnumber(int n) {
      numerator = n;
      denominator = 1;
  }

  int getNumerator() {
      return numerator;
  }

  int getDenominator() {
      return denominator;
  }

  void setNumerator(int n) {
      numerator = n;
  }

  void setDenominator(int n) {
      denominator = n;
  }

  static vector split(const string &s, char delim) {
      stringstream stream(s);
      string item;
      vector tokens;
      while (getline(stream, item, delim)) {
          tokens.push_back(item);
      }
      return tokens;
  }

  friend void operator >> (istream &input, rationalnumber &r);

  friend ostream& operator << (ostream &output, rationalnumber &r);

  rationalnumber operator + (rationalnumber &r) {
      int td,rd,num,den;
      td = this->denominator;
      num = this->numerator;
      den = td;
      rd = r.getDenominator();
      if (td!=rd) {
          num *= rd;
          den *= rd;
          num += td*r.getNumerator();
      }
      else {
          num += r.getNumerator();
      }
      return rationalnumber(num,den);
  }

  rationalnumber operator - (rationalnumber &r) {
      int td,rd,num,den;
      td = this->denominator;
      num = this->numerator;
      den = td;
      rd = r.getDenominator();
      if (td!=rd) {
          num *= rd;
          den *= rd;
          num -= td*r.getNumerator();
      }
      else {
          num -= r.getNumerator();
      }
      return rationalnumber(num,den);
  }

  rationalnumber operator * (rationalnumber &r) {
      return rationalnumber(this->numerator*r.getNumerator(),
                            this->denominator*r.getDenominator());
  }

  rationalnumber operator / (rationalnumber &r) {
      return rationalnumber(this->numerator*r.getDenominator(),
                            this->denominator*r.getNumerator());
  }

  bool operator == (rationalnumber &r) {
      return this->numerator*r.getDenominator()==this->denominator*r.getNumerator();
  }

  bool operator != (rationalnumber &r) {
      return !(this==&r);
  }

  bool operator > (rationalnumber &r) {
      return this->numerator*r.getDenominator()>this->denominator*r.getNumerator();
  }

  bool operator >= (rationalnumber &r) {
      return this == &r or this > &r;
  }

  bool operator < (rationalnumber &r) {
      return this->numerator*r.getDenominator()denominator*r.getNumerator();
  }

  bool operator <= (rationalnumber &r) {
      return this == &r or this < &r;
  }
};
